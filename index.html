Linguagem Potigol

Programação para todos

Potigol é ...

Uma linguagem moderna para aprender a programar.

Versão 0.9.2 Download

Características

  • Projetada para ser usada por alunos iniciantes
  • Tipagem estática com inferência de tipos
  • Palavras-chave em português
  • Multiparadigma
  • Estímulo ao paradigma funcional: valores imutáveis, casamento de padrões, funções como valores

A Linguagem

Variáveis

x = 10        # Valor fixo

var y = 10    # Valor alterável
y := y + 2

Entrada

a = leia_inteiro
b = leia_numero
c = leia_texto
x, y, z = leia_inteiro

Saída

escreva "Olá Mundo"   # Escreve e passa para a próxima linha
imprima "Olá "        # Escreve e continua na mesma linha
escreva "Mundo"

nome = "Mundo"
escreva "Olá {nome}"

se x > 5 então

se x > 5 então
   escreva "Maior do que cinco."
senão
  escreva "Menor ou igual a cinco."

  escreva "Maior do que seis."

  escreva "Maior do que quatro."

  escreva "Maior do que dois."

Escolha

escolha x
  caso 1 => escreva "Um"
  caso 2 => escreva "Dois"
  caso 3 => escreva "Três"
  caso _ => escreva "Outro valor"
fim

Para

para i de 1 até 10 faça
  escreva i
fim

Enquanto

var i = 0
enquanto i<=10 faça
  escreva i
  i := i + 1
fim

Funções

soma(x, y: Inteiro) = x + y

a, b = leia_inteiro
c = soma(a,b)
escreva "{a} + {b} = {c}"


fatorial(n: Inteiro)
  var f = 1
  para i de 2 até n faça
    f := f * i
  fim
  f
fim

a = leia_inteiro
escreva "Fatorial de {a} é {fatorial(a)}"
